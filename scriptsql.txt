CREATE DATABASE agendamaster DEFAULT CHARACTER SET utf8 DEFAULT COLLATE utf8_general_ci;
USE agendamaster;
CREATE TABLE agendamaster(
    id INT NOT NULL AUTO_INCREMENT,
    nome VARCHAR(45) NOT NULL,
    email VARCHAR(45) NOT NULL,
    cpf VARCHAR(45) NOT NULL,
    telefone VARCHAR(15) DEFAULT NULL,
    PRIMARY KEY(id)
);
